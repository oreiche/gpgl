1 /**
  2  * Creates wrapper for general purpose computing using WebGL.
  3  * @class GPGL Wrapper
  4  * @param {HTMLCanvasElement} canvas HTML5 canvas object.
  5  */
  6 function GPGL(canvas) {
  7     "use strict";
  8 
  9     /** @private */
 10 
 11     var gl = canvas.getContext("experimental-webgl");
 12     if (!gl) {
 13         throw "WebGL is not available";
 14     }
 15 
 16     var has_float = gl.getExtension("OES_texture_float");
 17 
 18     var gpgl = {
 19     /** @lends GPGL.prototype */
 20         /** @public */
 21 
 22         /**
 23          * WebGL Context
 24          * @type WebGLRenderingContext
 25          */
 26         gl: gl,
 27 
 28         /**
 29          * Indicates whether floating point textures extension is available.
 30          * @type Boolean
 31          */
 32         has_float: has_float,
 33 
 34         /**
 35          * Enumerator for image format
 36          * @class Image formats
 37          */
 38         Format: {
 39             /**
 40              * Image format for four channel unsigned byte.
 41              * @type Enum
 42              */
 43             UBYTE8888: 0,
 44 
 45             /**
 46              * Image format for single channel floating point.
 47              * @type Enum
 48              */
 49             FLOAT32:   1
 50         },
 51 
 52         /**
 53          * Enumerator for kernel argument type.
 54          * @class Kernel argument types
 55          */
 56         Arg: {
 57             /**
 58              * Kernel argument type floating point.
 59              * @type Enum
 60              */
 61             FLOAT: 0,
 62 
 63             /**
 64              * Kernel argument type integer.
 65              * @type Enum
 66              */
 67             INT:   1,
 68         },
 69 
 70         /**
 71          * Creates kernel function.
 72          * @param {String} source Source code of kernel function.
 73          * @returns {GPGL.Kernel}
 74          */
 75         createKernel: function(source) {
 76             /** @private */
 77 
 78             var texIds = {
 79                 list: [],
 80                 dict: {}
 81             }
 82 
 83             var program = gl.createProgram();
 84             this.program = program;
 85 
 86             var buffer = gl.createBuffer();
 87             gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
 88             gl.bufferData(gl.ARRAY_BUFFER,
 89                           new Float32Array([-1, -1,
 90                                              1, -1,
 91                                             -1,  1,
 92                                             -1,  1,
 93                                              1, -1,
 94                                              1,  1]),
 95                           gl.STATIC_DRAW);
 96 
 97             var v = gl.createShader(gl.VERTEX_SHADER),
 98                 f = gl.createShader(gl.FRAGMENT_SHADER);
 99 
100             gl.shaderSource(v, "precision highp float; \
101                                 attribute vec2 a_position; \
102                                 uniform float u_flipY; \
103                                 uniform vec2 global_size_abs; \
104                                 varying vec2 global_id_abs; \
105                                 varying vec2 global_id_norm; \
106                                 void main() { \
107                                     global_id_norm = a_position * 0.5 + 0.5; \
108                                     global_id_abs = global_size_abs * global_id_norm; \
109                                     gl_Position = vec4(a_position * vec2(1, u_flipY), 1.0, 1.0); \
110                                 }");
111             gl.shaderSource(f, "precision highp float; \
112                                 uniform vec2 global_size_abs; \
113                                 varying vec2 global_id_abs; \
114                                 varying vec2 global_id_norm; \
115                                 float shift_right(float v, float amt) {\
116                                   v = floor(v) + 0.5;\
117                                   return floor(v / exp2(amt));\
118                                 }\
119                                 float shift_left(float v, float amt) {\
120                                   return floor(v * exp2(amt) + 0.5);\
121                                 }\
122                                 \
123                                 float mask_last(float v, float bits) {\
124                                   return mod(v, shift_left(1.0, bits));\
125                                 }\
126                                 float extract_bits(float num, float from, float to) {\
127                                   from = floor(from + 0.5);\
128                                   to = floor(to + 0.5);\
129                                   return mask_last(shift_right(num, from), to - from);\
130                                 }\
131                                 vec4 encode_float(float val) {\
132                                   if (val == 0.0)\
133                                     return vec4(0, 0, 0, 0);\
134                                   float sign = val > 0.0 ? 0.0 : 1.0;\
135                                   val = abs(val);\
136                                   float exponent = floor(log2(val));\
137                                   float biased_exponent = exponent + 127.0;\
138                                   float fraction = ((val / exp2(exponent)) - 1.0) * 8388608.0;\
139                                   \
140                                   float t = biased_exponent / 2.0;\
141                                   float last_bit_of_biased_exponent = fract(t) * 2.0;\
142                                   float remaining_bits_of_biased_exponent = floor(t);\
143                                   \
144                                   float byte4 = extract_bits(fraction, 0.0, 8.0) / 255.0;\
145                                   float byte3 = extract_bits(fraction, 8.0, 16.0) / 255.0;\
146                                   float byte2 = (last_bit_of_biased_exponent * 128.0 + extract_bits(fraction, 16.0, 23.0)) / 255.0;\
147                                   float byte1 = (sign * 128.0 + remaining_bits_of_biased_exponent) / 255.0;\
148                                   return vec4(byte4, byte3, byte2, byte1);\
149                                 }" +
150                                 source);
151 
152             gl.compileShader(v);
153             if (!gl.getShaderParameter(v, gl.COMPILE_STATUS)) {
154                 throw "Failed to compile vertex shader: " +
155                       gl.getShaderInfoLog(v);
156             }
157 
158             gl.compileShader(f);
159             if (!gl.getShaderParameter(f, gl.COMPILE_STATUS)) {
160                 throw "Failed to compile kernel source: " +
161                       gl.getShaderInfoLog(f);
162             }
163 
164             gl.attachShader(program, v);
165             gl.attachShader(program, f);
166             gl.linkProgram(program);
167             if (!gl.getProgramParameter(program, gl.LINK_STATUS)) {
168                 throw "Failed to link program";
169             }
170 
171             var position = gl.getAttribLocation(program, "a_position");
172             var flip = gl.getUniformLocation(program, "u_flipY");
173             var size = gl.getUniformLocation(program, "global_size_abs");
174             var vp = gl.getParameter(gl.VIEWPORT);
175             gl.enableVertexAttribArray(position);
176             gl.vertexAttribPointer(position, 2, gl.FLOAT, false, 0, 0);
177 
178             var Target = {
179                 None:    0,
180                 Canvas:  1,
181                 Texture: 2
182             };
183 
184             var renderTarget = Target.None;
185 
186             var args = {};
187 
188             function setArg(name, type, dim, data, isarray) {
189                 if (dim !== data.length) {
190                     throw "Vector input data mismatch: " +
191                           dim + " !== " + data.length;
192                 }
193 
194                 var location, uniform;
195                 if (args[name]) {
196                     location = args[name].location;
197                     uniform = args[name].uniform;
198                 } else {
199                     location = gl.getUniformLocation(program, name);
200                     uniform = "uniform";
201                     switch (type) {
202                     case gpgl.Arg.INT:
203                         uniform += "" + dim + "i";//v";
204                         break;
205                     case gpgl.Arg.FLOAT:
206                         uniform += "" + dim + "f";//v";
207                         break;
208                     default:
209                         throw "Unknown argument type id '" + type + "'";
210                         break;
211                     }
212 
213                     if (isarray) {
214                         uniform += "v";
215                     }
216                 }
217 
218                 //gl[uniform](location, data);
219                 args[name] = {uniform: uniform, location: location, data: data, new: true};
220                 //gl[uniform](location, data[0], data[1], data[2], data[3]);
221             }
222 
223             /** @lends GPGL.Kernel.prototype */
224             return {
225                 /** @public */
226 
227                 /**
228                  * Set scalar kernel argument.
229                  * @param {String}   name  Name of the kernel uniform to map the value to.
230                  * @param {GPGL#Arg} type  Type of the kernel uniform.
231                  * @param {Number}   value Value to store in the kernel uniform.
232                  */
233                 setArgScalar: function(name, type, value) {
234                     setArg(name, type, 1, [value]);
235                 },
236 
237                 /**
238                  * Set vector kernel argument.
239                  * @param {String}   name Name of the kernel uniform to map the values to.
240                  * @param {GPGL#Arg} type Type of the kernel uniform values.
241                  * @param {Number}   dim  Dimension of the data array.
242                  * @param {Array}    data Data array containing the values to store.
243                  */
244                 setArgVector: function(name, type, dim, data) {
245                     setArg(name, type, dim, data);
246                 },
247 
248                 /**
249                  * Set array kernel argument.
250                  * @param {String}   name Name of the kernel uniform to map the array to.
251                  * @param {GPGL#Arg} type Type of the kernel uniform array.
252                  * @param {Array}    ref  Reference of the array to set.
253                  */
254                 setArgArray: function(name, type, ref) {
255                     setArg(name, type, 1, [ref], true);
256                 },
257 
258                 /**
259                  * Set image kernel argument.
260                  * @param {String}       name Name of the kernel sampler2D to map the image to.
261                  * @param {GPGL.Texture} tex  Texture object to set.
262                  */
263                 setArgImage: function(name, tex) {
264                     var id, texId = 0;
265                     if (texIds.dict[name] === undefined) {
266                         while (texIds.list[texId]) {
267                             ++texId;
268                         }
269                         texIds.dict[name] = texId;
270                         texIds.list[texId] = name;
271                     } else {
272                         texId = texIds.dict[name];
273                     }
274 
275                     id = gl["TEXTURE" + texId];
276                     if (id === undefined) {
277                         throw "To many textures bound to this kernel: " + texId;
278                     }
279 
280                     setArg(name, gpgl.Arg.INT, 1, [texId]);
281 
282                     //gl.enable(tex.type);
283                     gl.activeTexture(id);
284                     gl.bindTexture(tex.type, tex.id);
285                 },
286 
287                 /**
288                  * Execute kernel function.
289                  * @param {GPGL.Texture} dest (OPTIONAL) Target texture to render to. The kernel
290                  *                            renders to bound HTMLCanvasElement if target is not
291                  *                            defined.
292                  */
293                 run: function(dest) {
294                     gl.useProgram(program);
295                     gl.bindBuffer(gl.ARRAY_BUFFER, buffer);
296                     gl.vertexAttribPointer(position, 2, gl.FLOAT, false, 0, 0);
297 
298                     if (dest) {
299                         if (dest.format === gpgl.Format.FLOAT32) {
300                             throw "Rendering to float textures not supported yet";
301                         }
302 
303                         if (renderTarget !== Target.Texture) {
304                             gl.uniform1f(flip, 1);
305                             renderTarget = Target.Texture;
306                         }
307                         gl.uniform2f(size, dest.width, dest.height);
308                         gl.bindFramebuffer(gl.FRAMEBUFFER, dest.fb);
309                         gl.viewport(0, 0, dest.width, dest.height);
310                     } else {
311                         if (renderTarget !== Target.Canvas) {
312                             gl.uniform1f(flip, -1.0);
313                             gl.uniform2f(size, vp[2], vp[3]);
314                             renderTarget = Target.Canvas;
315                         }
316                         gl.bindFramebuffer(gl.FRAMEBUFFER, null);
317                     }
318 
319                     for (var i in args) {
320                         if (args[i].new === true) {
321                             if (args[i].uniform === "uniform1fv") {
322                                 // FIXME: Just a temporarily chrome fix
323                                 gl.uniform1fv(args[i].location, args[i].data[0]);
324                             } else {
325                                 gl[args[i].uniform](args[i].location, args[i].data[0], args[i].data[1], args[i].data[2], args[i].data[3]);
326                             }
327                             args[i].new = false;
328                         }
329                     }
330 
331                     // draw
332                     gl.drawArrays(gl.TRIANGLES, 0, 6);
333                     gl.finish();
334 
335                     if (renderTarget === Target.Texture) {
336                         // restore viewport
337                         gl.viewport(vp[0], vp[1], vp[2], vp[3]);
338                     }
339                 },
340 
341                 /**
342                  * Deletes all internally created WebGL handles.
343                  */
344                 delete: function() {
345                     gl.deleteShader(v);
346                     gl.deleteShader(f);
347                     gl.deleteBuffer(buffer);
348                     gl.deleteProgram(program);
349                 }
350             };
351         },
352 
353         /**
354          * Creates two dimensional image object (texture).
355          * @param {Number}      width  Width of image element.
356          * @param {Number}      height Height of image element.
357          * @param {GPGL#Format} format Pixel format used for image.
358          * @param {Array}       data   (OPTIONAL) Initial data for image.
359          * @param {Boolean}     linear (OPTIONAL) Enable bilinear filtering.
360          * @returns {GPGL.Image2D}
361          */
362         createImage2D: function(width, height, format, data, linear) {
363             /** @private */
364 
365             var values, chan, type, fb, array,
366                 tex = gl.createTexture();
367 
368             gl.bindTexture(gl.TEXTURE_2D, tex);
369             //gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);
370             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.NEAREST);
371             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, (linear ? gl.LINEAR : gl.NEAREST));
372             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
373             gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
374 
375             //FIXME HANDLE IMAGE DATA
376             switch (format) {
377             case gpgl.Format.UBYTE8888:
378                 if (data && data.length !== width * height * 4) {
379                     throw "Image input data mismatch: " +
380                           data.length + " !== " + width + " * " + height + " * 4";
381                 }
382                 chan = gl.RGBA;
383                 type = gl.UNSIGNED_BYTE;
384                 array = "Uint8Array";
385                 break;
386 
387             case gpgl.Format.FLOAT32:
388                 if (!has_float) {
389                     throw "Float textures are not available";
390                 }
391                 if (data && data.length !== width * height) {
392                     throw "Image input data mismatch: " +
393                           data.length + " !== " + width + " * " + height;
394                 }
395                 chan = gl.LUMINANCE;
396                 type = gl.FLOAT;
397                 array = "Float32Array";
398                 break;
399             }
400 
401             if (data !== undefined && data !== null) {
402                 values = new window[array](data.length);
403                 for (var i = 0; i < data.length; ++i) {
404                     values[i] = data[i];
405                 }
406             } else {
407                 values = null;
408             }
409 
410             gl.texImage2D(gl.TEXTURE_2D, 0, chan, width, height, 0, chan, type, values);
411 
412             fb = gl.createFramebuffer();
413             gl.bindFramebuffer(gl.FRAMEBUFFER, fb);
414             gl.framebufferTexture2D(gl.FRAMEBUFFER, gl.COLOR_ATTACHMENT0, gl.TEXTURE_2D, tex, 0); 
415 
416             /** @lends GPGL.Image2D.prototype */
417             return {
418                 /** @public */
419 
420                 /**
421                  * Texture handle.
422                  * @type WebGLTexture
423                  */
424                 id: tex,
425 
426                 /**
427                  * Framebuffer handle.
428                  * @type WebGLFramebuffer
429                  */
430                 fb: fb,
431                 
432                 /**
433                  * Image width.
434                  * @type Number
435                  */
436                 width: width,
437 
438                 /**
439                  * Image height.
440                  * @type Number
441                  */
442                 height: height,
443 
444                 /**
445                  * Image format.
446                  * @type GPGL#Format
447                  */
448                 format: format,
449                 
450                 /**
451                  * Image WebGL texture type.
452                  * @type Enum
453                  */
454                 type: gl.TEXTURE_2D,
455 
456                 /**
457                  * Read pixels from image.
458                  * @param {Boolean} decode (OPTIONAL) Decode pixels to floating point array.
459                  * @returns {UInt8Array|Float32Array} Containing the pixel data.
460                  */
461                 readPixels: function(decode) {
462                     if (this.format === gpgl.Format.FLOAT32) {
463                         throw "Reading from float textures not supported yet";
464                     }
465 
466                     gl.bindFramebuffer(gl.FRAMEBUFFER, this.fb);
467                     var pixels = new Uint8Array(this.width * this.height * 4);
468                     gl.readPixels(0, 0, this.width, this.height, gl.RGBA, gl.UNSIGNED_BYTE, pixels);
469 
470                     if (decode) {
471                         pixels = new Float32Array(pixels.buffer);
472                     }
473 
474                     return pixels;
475                 },
476 
477                 /**
478                  * Deletes internally created WebGL handles.
479                  */
480                 delete: function() {
481                     gl.deleteFramebuffer(fb);
482                     gl.deleteTexture(tex);
483                 }
484             };
485         }
486     };
487 
488     return gpgl;
489 }
